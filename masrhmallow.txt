from marshmallow import Schema, fields, validate, EXCLUDE, INCLUDE, RAISE, pre_load, post_load, pre_dump, post_dump
import simplejson
from datetime import datetime

# üè¢ Subesquema para datos anidados (Ej: Tienda)
class StoreSchema(Schema):
    id = fields.Int(required=True)
    name = fields.Str(required=True, validate=validate.Length(min=3, max=50))

# üë§ Esquema Principal con TODAS las opciones
class UserSchema(Schema):
    # üîπ Campos b√°sicos
    id = fields.Int(dump_only=True)  # Se muestra, pero no se carga
    name = fields.Str(required=True, validate=validate.Length(min=2, max=100))
    email = fields.Email(required=True)
    password = fields.Str(load_only=True, validate=validate.Length(min=8))  # Solo para carga
    age = fields.Int(validate=validate.Range(min=18, max=100))  # Solo permite entre 18 y 100 a√±os
    active = fields.Bool(default=True)  # Campo booleano con valor por defecto
    balance = fields.Float(default=0.0)  # Valor flotante con valor inicial
    created_at = fields.DateTime(dump_only=True, format="%Y-%m-%d %H:%M:%S")  # Fecha formateada

    # üîπ Listas y Tuplas
    tags = fields.List(fields.Str(), validate=validate.Length(max=5))  # M√°ximo 5 etiquetas
    coordinates = fields.Tuple((fields.Float(), fields.Float()))  # Tupla con latitud/longitud

    # üîπ Campos anidados
    store = fields.Nested(StoreSchema)  # Un solo objeto
    stores = fields.List(fields.Nested(StoreSchema))  # Lista de objetos anidados

    # üîπ Extraer solo un campo de un esquema anidado
    store_name = fields.Pluck(StoreSchema, "name")

    # üîπ Validaciones avanzadas
    role = fields.Str(validate=validate.OneOf(["admin", "user", "guest"]))  # Solo permite estos valores
    preferences = fields.Dict(keys=fields.Str(), values=fields.Raw())  # Diccionario din√°mico

    class Meta:
        fields = ("id", "name", "email", "age", "active", "balance", "created_at", 
                  "tags", "coordinates", "store", "stores", "store_name", "role", "preferences")
        exclude = ("password",)  # Excluye expl√≠citamente "password"
        ordered = True  # Mantiene el orden de los campos
        unknown = EXCLUDE  # Ignora campos desconocidos
        datetimeformat = "%Y-%m-%d %H:%M:%S"  # Formato para fechas
        render_module = simplejson  # Usa `simplejson` en vez de `json`
        index_errors = True  # Muestra errores con √≠ndices en listas

    # üîπ M√©todos de transformaci√≥n de datos
    @pre_load
    def process_input(self, data, **kwargs):
        """ Convierte strings a min√∫sculas antes de la carga """
        if "name" in data:
            data["name"] = data["name"].strip().lower()
        return data

    @post_load
    def create_user(self, data, **kwargs):
        """ Convierte datos en un objeto despu√©s de cargar """
        data["created_at"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        return data

    @pre_dump
    def before_dump(self, data, **kwargs):
        """ Modifica datos antes de la serializaci√≥n """
        data["name"] = data["name"].title()  # Capitaliza el nombre
        return data

    @post_dump
    def after_dump(self, data, **kwargs):
        """ Modifica datos despu√©s de la serializaci√≥n """
        data["serialized_at"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        return data

# üìå Ejemplo de uso

from sqlalchemy import (
    Column, Integer, String, Boolean, Float, DateTime, ForeignKey, Table, create_engine
)
from sqlalchemy.orm import relationship, sessionmaker, declarative_base
from marshmallow import Schema, fields, validate, pre_load, post_load, pre_dump, post_dump
from datetime import datetime

# üèõ Base de datos y ORM
Base = declarative_base()

# üîó Tabla intermedia para relaci√≥n Many-to-Many (Usuarios <-> Roles)
user_roles = Table(
    "user_roles",
    Base.metadata,
    Column("user_id", Integer, ForeignKey("users.id"), primary_key=True),
    Column("role_id", Integer, ForeignKey("roles.id"), primary_key=True),
)

# üè¨ Modelo SQLAlchemy: Tienda
class Store(Base):
    __tablename__ = "stores"
    id = Column(Integer, primary_key=True)
    name = Column(String(50), nullable=False)

# üë§ Modelo SQLAlchemy: Usuario
class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True)
    name = Column(String(100), nullable=False)
    email = Column(String(100), nullable=False, unique=True)
    password = Column(String(255), nullable=False)
    age = Column(Integer, nullable=True)
    active = Column(Boolean, default=True)
    balance = Column(Float, default=0.0)
    created_at = Column(DateTime, default=datetime.utcnow)
    
    store_id = Column(Integer, ForeignKey("stores.id"), nullable=True)  # Relaci√≥n One-to-Many
    store = relationship("Store", backref="users")

    roles = relationship("Role", secondary=user_roles, backref="users")  # Many-to-Many

# üé≠ Modelo SQLAlchemy: Roles
class Role(Base):
    __tablename__ = "roles"
    id = Column(Integer, primary_key=True)
    name = Column(String(50), unique=True)

# üé® Esquema de Marshmallow: Tienda
class StoreSchema(Schema):
    id = fields.Int(dump_only=True)
    name = fields.Str(required=True, validate=validate.Length(min=3, max=50))

# üë§ Esquema de Marshmallow: Usuario
class UserSchema(Schema):
    id = fields.Int(dump_only=True)
    name = fields.Str(required=True, validate=validate.Length(min=2, max=100))
    email = fields.Email(required=True)
    password = fields.Str(load_only=True, validate=validate.Length(min=8))
    age = fields.Int(validate=validate.Range(min=18, max=100))
    active = fields.Bool(default=True)
    balance = fields.Float(default=0.0)
    created_at = fields.DateTime(dump_only=True, format="%Y-%m-%d %H:%M:%S")

    store = fields.Nested(StoreSchema, allow_none=True)
    roles = fields.List(fields.Str())  # Convertir√° `roles` a una lista de nombres

    class Meta:
        ordered = True
        include_relationships = True  # Serializa relaciones autom√°ticamente

    @pre_load
    def process_input(self, data, **kwargs):
        """ Convierte nombres en min√∫sculas antes de la carga """
        if "name" in data:
            data["name"] = data["name"].strip().lower()
        return data

    @post_load
    def create_user(self, data, **kwargs):
        """ Agrega fecha de creaci√≥n si no est√° presente """
        if "created_at" not in data:
            data["created_at"] = datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")
        return data

    @pre_dump
    def before_dump(self, data, **kwargs):
        """ Capitaliza nombres antes de la serializaci√≥n """
        if hasattr(data, "name"):
            data.name = data.name.title()
        return data

    @post_dump
    def after_dump(self, data, **kwargs):
        """ Agrega campo de serializaci√≥n """
        data["serialized_at"] = datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")
        return data

# üî• Configurar Base de Datos (SQLite en memoria)
engine = create_engine("sqlite:///:memory:", echo=False)
Base.metadata.create_all(engine)

# üõ† Crear sesi√≥n
Session = sessionmaker(bind=engine)
session = Session()

# üìå Insertar datos de prueba
store = Store(name="Super Tienda")
role_admin = Role(name="admin")
role_user = Role(name="user")
user = User(name="Juan P√©rez", email="juan@example.com", password="secreto123", age=30, store=store, roles=[role_admin, role_user])
session.add_all([store, role_admin, role_user, user])
session.commit()

# üîç Consultar usuario y serializar con `marshmallow`
user_data = session.query(User).first()
schema = UserSchema()
serialized_data = schema.dump(user_data)

print("üîπ Usuario Serializado:", serialized_data)


Explicaci√≥n Detallada
‚úî Modelos de SQLAlchemy
Usuario (User) con relaci√≥n:
One-to-Many con Store
Many-to-Many con Role
Tienda (Store)
Rol (Role) usado en relaci√≥n Many-to-Many
‚úî Opciones en Meta
ordered = True: Mantiene el orden de los campos
include_relationships = True: Serializa relaciones autom√°ticamente
‚úî Transformaciones con @pre_load, @post_load, @pre_dump, @post_dump
@pre_load: Convierte nombres a min√∫sculas
@post_load: Agrega created_at si falta
@pre_dump: Capitaliza nombres
@post_dump: Agrega timestamp de serializaci√≥n
‚úî Relaciones en Marshmallow
fields.Nested(StoreSchema) para relaciones One-to-Many
fields.List(fields.Str()) para convertir Many-to-Many a una lista de nombres
